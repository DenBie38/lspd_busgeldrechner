import requests from bs4 import BeautifulSoup # URL der Seite, von der du die Daten abrufen willst url = 'https://gta5grand.com/forum/threads/947959/' # HTTP-Request, um die Seite zu bekommen response = requests.get(url) # Überprüfen, ob die Seite erfolgreich geladen wurde if response.status_code == 200: page_content = response.text soup = BeautifulSoup(page_content, 'html.parser') # Suche nach den spezifischen Daten auf der Seite (angepasst an das Forum) # Zum Beispiel: Suche nach Tabellen oder Absätzen, die Bußgeldinformationen enthalten violations = soup.find_all('div', class_='post-text') # Beispiel für Forum-Post-Inhalte for violation in violations: print(violation.text.strip()) # Gibt die Strafen/Verstöße aus else: print("Fehler beim Laden der Seite!")
